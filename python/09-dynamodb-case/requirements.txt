"""
WebSocket断开连接处理 - $disconnect路由
处理WebSocket客户端断开连接请求
"""
import sys
import os

# 添加shared目录到Python路径
sys.path.insert(0, os.path.join(os.path.dirname(__file__), '..', 'shared'))

from context_manager import ContextManager

# 初始化上下文管理器
context_mgr = ContextManager()


def lambda_handler(event, context):
    """
    处理WebSocket断开连接请求

    Args:
        event: API Gateway事件
        context: Lambda上下文

    Returns:
        dict: 响应对象
    """
    try:
        connection_id = event.get('requestContext', {}).get('connectionId')

        if connection_id:
            # 清理该连接的上下文数据
            context_mgr.clear_context(connection_id)
            print(f"Connection closed: {connection_id}")

        return {
            'statusCode': 200,
            'body': 'Disconnected'
        }

    except Exception as e:
        print(f"Error in disconnect handler: {e}")
        return {
            'statusCode': 200,  # 即使出错也返回200，避免重试
            'body': 'Disconnected'
        }
